<body>

  <h1>Car Booking Platform</h1>

  <div class="car-list" id="carList">
    <!-- Cars will be rendered here -->
  </div>

  <div class="booking-form">
    <h2>Book Your Car</h2>
    <label for="carSelect">Select Car:</label>
    <select id="carSelect"></select>

    <label for="date">Booking Date:</label>
    <input type="date" id="date" required />

    <label for="days">Number of Days:</label>
    <input type="number" id="days" min="1" value="1" required />

    <button onclick="bookCar()">Book Now</button>
  </div>

  <script>
    const cars = [
      { id: 1, name: "Sedan", pricePerDay: 40, img: "https://via.placeholder.com/200x120?text=Sedan" },
      { id: 2, name: "SUV", pricePerDay: 60, img: "https://via.placeholder.com/200x120?text=SUV" },
      { id: 3, name: "Convertible", pricePerDay: 80, img: "https://via.placeholder.com/200x120?text=Convertible" }
    ];

    const carList = document.getElementById("carList");
    const carSelect = document.getElementById("carSelect");

    // Render car options and car cards
    cars.forEach(car => {
      // Add to display
      const carDiv = document.createElement("div");
      carDiv.className = "car";
      carDiv.innerHTML = `
        <img src="${car.img}" alt="${car.name}" />
        <h3>${car.name}</h3>
        <p>Price Per Day: $${car.pricePerDay}</p>
      `;
      carList.appendChild(carDiv);

      // Add to select dropdown
      const option = document.createElement("option");
      option.value = car.id;
      option.textContent = car.name;
      carSelect.appendChild(option);
    });

    function bookCar() {
      const selectedCarId = parseInt(carSelect.value);
      const date = document.getElementById("date").value;
      const days = parseInt(document.getElementById("days").value);

      if (!date) {
        alert("Please select a booking date.");
        return;
      }

      // Check if booking date is in the past
      const today = new Date().toISOString().split("T")[0];
      if (date < today) {
        alert("Please select a future date.");
        return;
      }

      if (days < 1) {
        alert("Number of days must be at least 1.");
        return;
      }

      const car = cars.find(c => c.id === selectedCarId);
      if (!car) {
        alert("Please select a car.");
        return;
      }

      const totalPrice = car.pricePerDay * days;

      alert(`Booking Confirmed!
Car: ${car.name}
Date: ${date}
Days: ${days}
Total Price: $${totalPrice}`);
    }
  </script>
